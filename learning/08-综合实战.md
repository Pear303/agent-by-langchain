```python
class LCAgent:
    def __init__(self, model="deepseek-v4-flash", max_iterations=50):
        self.root = Path(__file__).resolve().parent.parent

        sub_reg = SubagentRegistry(self.root / "templates" / "subagents", skills)
        set_subagent_deps(llm=self.llm, registry=sub_reg)

        # ==== 第4步：记忆系统 ====
        self.memory_store = MemoryStore(memory_dir=self.root / "memory", ...)
        self.token_tracker = TokenTracker(...)
        self.compactor = Compactor(openai_client, model, self.memory_store)

        # ==== 第5步：上下文构建 ====
        ctx = ContextBuilder(self.root / "templates", skills, memory=self.memory_store)
        system_prompt = ctx.build_system_prompt()

            ("system", system_prompt),

            ("placeholder", "{agent_scratchpad}"),

        # ==== 第7步：AgentExecutor ====
        self.executor = AgentExecutor(
            agent=agent, tools=self.tools,
            max_iterations=max_iterations,
            verbose=True,
            handle_parsing_errors=True,
            callbacks=[TokenCallback(self.token_tracker, model)],
        )

        # ==== 第8步：启动补归档 ====
        unarchived = self.memory_store.load_unarchived_history()
        if len(unarchived) >= 2:
            self.compactor.compact_startup(unarchived)
```

**运行阶段**——对话循环：

```python
class LCAgent:
    def run(self):
        """多轮对话循环"""
        print("🤖 智能助手启动。输入 'exit' 退出。")

        while True:
            user_input = input("You🫅 : ").strip()
            if user_input == "exit":
                break
            if not user_input:
                continue

            # 调用 AgentExecutor
            result = self.executor.invoke({
                "input": user_input,
                "chat_history": self.memory_store.messages,
            })
            reply = result["output"]

            # 持久化
            self.memory_store.append_history("user", user_input)
            self.memory_store.append_history("assistant", reply)

            # 检查压缩
            self._maybe_compact()

            # 输出
            print(f"智能助手🧰: {reply}")

    def _maybe_compact(self):
        if self.token_tracker.should_compact(
            max_context=200_000, threshold=0.7
        ):
            self.memory_store.messages = self.compactor.compact(
                self.memory_store.messages
            )
```

### 8.2.4 子代理并行执行：subagent_parallel.py

```python
class ParallelAgentExecutor(AgentExecutor):
    """支持工具级并发的 AgentExecutor"""

    _READ_ONLY_TOOLS = {"web_fetch", "read_file", "glob_tool", "grep_tool", "load_skill"}

    def _iter_next_step(self, *args, **kwargs):
        """重写 AgentExecutor 的迭代步骤"""
        for result in super()._iter_next_step(*args, **kwargs):
            if isinstance(result, AgentAction):
                # 收集同一帧的工具调用
                ...
            elif isinstance(result, AgentFinish):
                yield result

        # 分组并发执行只读工具
        for group in self._group_by_concurrency(actions):
            if len(group) == 1:
                yield self._perform_agent_action(group[0])
            else:
                with ThreadPoolExecutor(max_workers=len(group)) as pool:
                    results = list(pool.map(_run, group))
                for step in results:
                    yield step
```

### 8.2.5 记忆系统：memory.py

```python
class MemoryStore(BaseChatMessageHistory):
    messages: list[BaseMessage] = Field(default_factory=list)

    def __init__(self, memory_dir, user_file):
        self.memory_dir = Path(memory_dir)
        self.memory_dir.mkdir(parents=True, exist_ok=True)
        self.history_file = self.memory_dir / "history.jsonl"
        self.memory_file = self.memory_dir / "MEMORY.md"
        self.user_file = Path(user_file)
        self._load_history()

    def _load_history(self):
        """启动时从 JSONL 加载历史消息"""
        if not self.history_file.exists():
            return
        for line in self.history_file.read_text().splitlines():
            if not line.strip():
                continue
            entry = json.loads(line)
            role = _JSONL_ROLE_TO_TYPE.get(entry["role"], "human")
            cls = {"human": HumanMessage, "assistant": AIMessage, "system": SystemMessage}
            msg = cls[role](content=entry["content"])
            self.messages.append(msg)

    def add_messages(self, messages):
        """实现 BaseChatMessageHistory 接口"""
        for msg in messages:
            role = _TYPE_TO_JSONL_ROLE.get(msg.type, "unknown")
            extra = getattr(msg, "additional_kwargs", None) or None
            self.append_history(role, msg.content, additional_kwargs=extra)

    def read_memory(self):
        """读取长期记忆"""
        if self.memory_file.exists():
            return self.memory_file.read_text(encoding="utf-8")
        return ""
```

### 8.2.6 压缩器：compactor.py

```python
class Compactor:
    K = 10  # 保留最近轮数

    def compact(self, history):
        if len(history) <= self.K:
            return history

        old = history[:-self.K]
        self._run_compaction(old)
        return history[-self.K:]

    def _run_compaction(self, old_messages):
        prompt = _PROMPT_TEMPLATE.format(
            old_conversation=_messages_to_text(old_messages),
            current_memory=self.memory.read_memory(),
            current_user=self.memory.read_user(),
            today_episode=self.memory.read_today_episode(),
        )
        resp = self.client.chat.completions.create(
            model=self.model,
            messages=[{"role": "user", "content": prompt}],
        )
        text = resp.choices[0].message.content

        # 解析 <episode>、<updated_memory>、<updated_user>
        if episode := _extract("episode", text):
            self.memory.append_episode(episode)
        if new_memory := _extract("updated_memory", text):
            self.memory.write_memory(new_memory)
        if new_user := _extract("updated_user", text):
            self.memory.write_user(new_user)

        self.memory.append_compact_marker()
```

```python
@tool
def calculate(expression: str) -> str:
    """执行数学计算。支持 +, -, *, /, 括号等表达式。
    Args:
        expression: 数学表达式，如 "2 + 3 * 4"
    """
    try:
        # 安全计算（只允许数学表达式）

        if not all(c in allowed for c in expression):
            return "Error: 只允许数学表达式"
```

## 8.4 常见问题与排查

### Q1: Agent 总是调用错误的工具？

**可能原因**：
- 工具 `description` 写的不清楚
- 工具太多，LLM 混淆了
- 需要调整工具名称为更直观的名字

**解决方法**：
```python
# 不好的描述
@tool
def rf(path: str) -> str: ...

# 好的描述
@tool
def read_file(file_path: str, offset: int = 0, limit: int = 2000) -> str:
    """读取工作区文件的内容。支持 offset/limit 分页。"""
```

### Q2: Agent 陷入死循环？

**检查**：
1. `max_iterations` 是否设置合理
2. `handle_parsing_errors=True` 是否启用
3. 工具的返回值是否清晰（LLM 能否理解结果）

### Q3: 子代理没有按预期工作？

**检查**：
1. 身份模板写的是否清晰
2. 工具白名单是否包含所需工具
3. `max_turns` 是否足够
4. 子代理是用独立上下文启动的（`chat_history=[]`）

### Q4: 记忆没有持久化？

**检查**：
1. `memory/` 目录是否存在
2. `history.jsonl` 是否能正常写入
3. `MemoryStore` 是否实现了 `add_messages` 方法
4. 启动时 `_load_history()` 是否正确加载

## 8.5 扩展学习路线

掌握了本项目后，可以继续探索的方向：

### 方向1：LangGraph

LangGraph 是 LangChain 的"进化版"，专为更复杂的 Agent 工作流设计：

```
传统 AgentExecutor: 线性循环
LangGraph: 有向图（节点 + 边 + 条件跳转）

适用场景：
  - 需要严格的工作流控制
  - 复杂的多 Agent 通信
  - 人机协同（Human-in-the-loop）
```

本项目暂时没有使用 LangGraph 的复杂特性，如果你遇到：
- 多个 Agent 之间需要互相通信
- 流程不是线性的（有循环、分支、合并）
- 需要更精细的流程控制

那么 LangGraph 会是下一步学习的方向。

### 方向2：RAG（检索增强生成）

```
Agent + RAG = 既能推理，又能从知识库中查找信息

你可以：
  1. 把文档向量化存入数据库
  2. 给 Agent 添加检索工具
  3. Agent 在回答前先检索相关知识
```

### 方向3：MCP（Model Context Protocol）

MCP 是 AI Agent 工具调用的新兴标准协议：
- 标准化工具发现和调用
- 远程工具支持（通过 MCP 服务器）
- 动态工具注册

### 方向4：生产化

把 Agent 部署到生产环境需要考虑：
- 错误恢复
- 日志监控
- 并发请求管理
- 成本控制
- 安全审计
